geomesa = {
  sfts = {
    "custom-countries" = {
      attributes = [
        { name = "ISO",   type = "String",        index = false                              }
        { name = "NAME",  type = "String",        index = false                              }
        { name = "GEOM",  type = "MultiPolygon",  index = false, srid = 4326, default = true }
      ]
    }
    "custom-states" = {
      attributes = [
        { name = "NAME",          type = "String",        index = false                              }
        { name = "COUNTRY_ISO",   type = "String",        index = false                              }
        { name = "COUNTRY_NAME",  type = "String",        index = false                              }
        { name = "GEOM",          type = "MultiPolygon",  index = false, srid = 4326, default = true }
      ]
    }
    "custom-people" = {
      attributes = [
        { name = "pid",         type = "Integer",      index = false                              }
        { name = "name",        type = "String",       index = false                              }
        { name = "surname",     type = "String",       index = false                              }
        { name = "birthday",    type = "Date",         index = false                              }
        { name = "birth_geom",  type = "Point",        index = false, srid = 4326, default = true }
      ]
    }
  }
  converters = {
    "custom-countries-csv" = {
      type = "delimited-text",
      format = "CSV",
      id-field = "$name",
      fields = [
        { name = "iso",   transform = "$1::string"        }
        { name = "name",  transform = "$2::string"        }
        { name = "geom",  transform = "multipolygon($3)"  }
      ]
    }
    "custom-states-csv" = {
      type = "delimited-text",
      format = "CSV",
      id-field = "$name",
      fields = [
        { name = "name",          transform = "$1::string"        }
        { name = "country_iso",   transform = "$2::string"        }
        { name = "country_name",  transform = "$3::string"        }
        { name = "geom",          transform = "multipolygon($4)"  }
      ]
    }
    "custom-people-csv" = {
      type = "delimited-text",
      format = "CSV",
      options {
        skip-lines = 1
      },
      id-field = "toString($pid)",
      fields = [
        { name = "pid",         transform = "$1::int"                       }
        { name = "name",        transform = "$2::string"                    }
        { name = "surname",     transform = "$3::string"                    }
        { name = "birthday",    transform = "date('yyyy-MM-dd', $4)"        }
        { name = "birth_lat",   transform = "$5::double"                    }
        { name = "birth_lon",   transform = "$6::double"                    }
        { name = "birth_geom",  transform = "point($birth_lon, $birth_lat)" }
      ]
    }
  }
}
