SELECT 
    SUM(L.LO_EXTENDEDPRICE*L.LO_DISCOUNT) AS REVENUE 
FROM 
    LINEORDER L, 
    DATE D, 
    CUSTOMER C, 
    SUPPLIER S
WHERE 
    L.LO_ORDERDATE = D.D_DATEKEY AND 
    L.LO_SUPPKEY = S.S_SUPPKEY AND 
    L.LO_CUSTKEY = C.C_CUSTKEY AND 
    D.D_YEAR = 1993 AND 
    L.LO_DISCOUNT BETWEEN 1 AND 3 AND 
    L.LO_QUANTITY < 25 AND 
    INTERSECTS(C.C_REGION, QW1) AND 
    INTERSECTS(C.C_NATION, QW2) AND 
    INTERSECTS(C.C_CITY, QW3) AND 
    INTERSECTS(C.C_STREET, QW4) AND 
    INTERSECTS(C.C_ADDRESS, QW5);
