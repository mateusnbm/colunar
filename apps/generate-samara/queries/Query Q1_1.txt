SELECT SUM(LO_EXTENDEDPRICE*LO_DISCOUNT) AS REVENUE FROM LINEORDER, DATE, CUSTOMER, SUPPLIER WHERE LO_ORDERDATE = D_DATEKEY AND LO_SUPPKEY = S_SUPPKEY AND LO_CUSTKEY = C_C_CUSTKEY AND D_YEAR = 1993 AND LO_DISCOUNT BETWEEN 1 AND 3 AND LO_QUANTITY < 25 AND INTERSECTS(REGION, QW1)AND INTERSECTS(NATION, QW2)AND INTERSECTS(CITY, QW3)AND INTERSECTS(STREET, QW4)AND INTERSECTS(ADDRESS, QW5)AND INTERSECTS(EMPTY AREA, QW6)
