SELECT SUM(LO_EXTENDEDPRICE*LO_DISCOUNT) AS REVENUE FROM LINEORDER, DATE, CUSTOMER, SUPPLIER WHERE LO_ORDERDATE = D_DATEKEY AND LO_SUPPKEY = S_SUPPKEY AND LO_CUSTKEY = C_C_CUSTKEY AND D_WEEKNUMINYEAR = 6 AND D_YEAR = 1994 AND LO_DISCOUNT BETWEEN 5 AND 7 AND LO_QUANTITY BETWEEN 26 AND 35 AND INTERSECTS(REGION, QW1)AND INTERSECTS(NATION, QW2)AND INTERSECTS(CITY, QW3)AND INTERSECTS(STREET, QW4)AND INTERSECTS(ADDRESS, QW5)AND INTERSECTS(EMPTY AREA, QW6)
